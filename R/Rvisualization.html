  geom_point(aes(col=state, size=popdensity)) +   # draw points
  geom_smooth(method="loess", se=F) + 
  xlim(c(0, 0.1)) + 
  ylim(c(0, 500000)) +   # draw smoothing line

       x="Area", 
       title="Scatterplot + Encircle", 
       caption="Source: midwest")
